-- ============================================================ -- SYSTÈME DE TICKETS — SQL à coller dans Supabase SQL Editor -- ============================================================ -- À exécuter une seule fois. Crée le fil de discussion des -- tickets (ticket_messages) et met en place les règles RLS. -- ============================================================ -- 1) TABLE DU FIL DE DISCUSSION -- Chaque ligne = un message dans un ticket (soit la réponse -- de l'admin, soit un message du client). create table if not exists public.ticket_messages ( id uuid primary key default gen_random_uuid(), ticket_id uuid not null references public.tickets(id) on delete cascade, sender_role text not null, -- 'admin' ou 'client' sender_name text, -- nom affiché de l'expéditeur body text not null, -- contenu du message created_at timestamptz not null default now() ); -- Index pour charger rapidement le fil d'un ticket create index if not exists ticket_messages_ticket_id_idx on public.ticket_messages (ticket_id, created_at); -- ============================================================ -- 2) RLS SUR LA TABLE tickets -- ============================================================ alter table public.tickets enable row level security; -- On repart propre : on supprime d'éventuelles policies existantes drop policy if exists "tickets_admin_all" on public.tickets; drop policy if exists "tickets_client_select" on public.tickets; drop policy if exists "tickets_client_insert" on public.tickets; -- L'admin voit et gère TOUS les tickets create policy "tickets_admin_all" on public.tickets for all using ( exists ( select 1 from public.profiles p where p.id = auth.uid() and p.role = 'admin' ) ) with check ( exists ( select 1 from public.profiles p where p.id = auth.uid() and p.role = 'admin' ) ); -- Le client voit uniquement les tickets de SON hôtel create policy "tickets_client_select" on public.tickets for select using ( hotel_id = ( select p.hotel_id from public.profiles p where p.id = auth.uid() ) ); -- Le client peut créer un ticket pour SON hôtel create policy "tickets_client_insert" on public.tickets for insert with check ( hotel_id = ( select p.hotel_id from public.profiles p where p.id = auth.uid() ) ); -- ============================================================ -- 3) RLS SUR LA TABLE ticket_messages -- ============================================================ alter table public.ticket_messages enable row level security; drop policy if exists "tmsg_admin_all" on public.ticket_messages; drop policy if exists "tmsg_client_select" on public.ticket_messages; drop policy if exists "tmsg_client_insert" on public.ticket_messages; -- L'admin lit et écrit tous les messages create policy "tmsg_admin_all" on public.ticket_messages for all using ( exists ( select 1 from public.profiles p where p.id = auth.uid() and p.role = 'admin' ) ) with check ( exists ( select 1 from public.profiles p where p.id = auth.uid() and p.role = 'admin' ) ); -- Le client lit les messages des tickets de son hôtel create policy "tmsg_client_select" on public.ticket_messages for select using ( exists ( select 1 from public.tickets t join public.profiles p on p.id = auth.uid() where t.id = ticket_messages.ticket_id and t.hotel_id = p.hotel_id ) ); -- Le client peut ajouter un message à un ticket de son hôtel create policy "tmsg_client_insert" on public.ticket_messages for insert with check ( exists ( select 1 from public.tickets t join public.profiles p on p.id = auth.uid() where t.id = ticket_messages.ticket_id and t.hotel_id = p.hotel_id ) ); -- ============================================================ -- FIN. Après exécution, la table ticket_messages est prête -- et les tickets sont correctement cloisonnés par hôtel. -- ============================================================
